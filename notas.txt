---------------------SOBRE GIT COMANDOS ------------------------
//abrimos GIT BASH HERE en la carpeta donde queemos hacer respaldo
git init  //para q inicialice y crea carpeta
git status //para ver como esta 
git status -s // mas especifico que muestra las carpetas y archivos de momentos en la carpeta del proyecto, cuando aparece ?, esntonces no hay seguimiento de estos files
// es decir no se ha creado copias de respaldo ejem: ?? css/

git add .   //para agregar todo, con vs podemos ir al tecer icono para para darle commit a todo o solo un archivo no mas
            //sigo menos en vs es quitar del respaldo de GIT, 
            //donde dice boton COMMIT: para darle respaldo a todos archivos y guardarlo en ensayo
git add index.html //agregamos el archivo con el nomre mas su extencion si html...
            //A  index.html : este archivo esta en el area de ensayo, lo esta monitoreando o haciendo el seguimiento
            //M index.html : ha sido modificado y no hay respaldo que se hayay echo que esta en rojo
            //M  index.html :  el archivo ha sido modificado, verde y ha tomado nota de la modificacion que ha echo
            //AM css/estilos.css , modificado este file A de color verde y M de rojo
            // ejem no sale el nombre index.html , porque ahora mismo esta restaurado

//ADD despues COMMIT
git commit -m "Se agrega los cambios" //es un commit, caso que sale error ponemos el email y la contra despues agregamos otra vez el commit
git commit -m "Comienzo del proyecto" //1 file changed, 13 insertions(+), va salir asi cuando se agrego correctamente el commit, es decir un respaldo 
git commit -am "Parrafo y tamaño de fuente y css" //add y comit juatos a la vez
